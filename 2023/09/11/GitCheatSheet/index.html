设置与配置

# 初始化一个新的仓库
git init
# 克隆并创建远程仓库的本地副本
git clone <url>
# 配置全局设置
git config --global <setting_name> <value>
# 为指定的仓库配置本地设置
git config --local <setting_name> <value>

# --------------- Advanced ------------------

# 显示配置的设置的概要
git config --list
# 给消息设置自定义文本编辑器
git config --global core.editor "<editor_command>"
# 创建命令别名
git config --global alias.<shortcut> <command>
# 开启输出的自动着色
git config --global color.ui auto
# 缓存一段时间的凭证
git config --global credential.helper 'cache --timeout=<seconds>'
# 配置以检测特定类型的空白错误
git config --global core.whitespace <options>
# 获取更新时自动修剪远程追踪分支
git config --global fetch.prune true
# 设置自定义diff工具
git config --global diff.tool <tool>
# 设置自定义合并工具
git config --global merge.tool <tool>
# 用自定义diff工具比较修改
git difftool
# 用自定义合并工具解决合并冲突
git mergetool

文件操作

# 展示工作树状态
git status
# 添加文件到暂存区
git add <file(s)>
# 从工作树和暂存区删除文件
git rm <file(s)>
# 移动或重命名文件
git mv <old_file> <new_file>
# 提交带有消息的更改
git commit -m "commit message"
# 显示工作数和上次提交之间的差别
git diff

# --------------- Advanced ------------------

# 假定追踪的文件没有修改
git update-index --assume-unchanged <file>
# 恢复追踪的修改的正常行为
git update-index --no-assume-unchanged <file>
# 显示两次提交之间的差异
git diff <commit_id1>..<commit_id2>
# 从暂存区移除文件，但保留在工作目录中
git rm --cached <file_name>

分支与合并

# 列出分支
git branch
# 创建一个新的分支
git branch <branch_name>
# 切换到指定分支
git checkout <branch_name>
# 将分支合并到当前分支
git merge <branch_name>
# 删除指定分支
git branch -d <branch_name>
# 列出远程分支
git branch -r

# --------------- Advanced ------------------

# 列出带有附近信息的分支
git branch -vv
# 基于远程分支创建新的分支
git checkout -b <branch_name> <remote_name>/<remote_branch>
# 产生冲突时取消合并
git merge --abort
# 将当前分支重新定位到另一个分支上
git rebase <branch_name>
# 取消正在进行的重置操作
git rebase --abort
# 可编辑、压缩、重新排序或删除提交的交互式重置
git rebase -i
# 以交互的方式将当前分支的提交重置到远程分支上
git rebase -i <remote_name>/<remote_branch>

远程仓库

# 列出远程仓库
git remote
# 添加远程仓库
git remote add <name> <url>
# 从远程仓库中获取
git fetch <remote_name>
# 从远程分支拉取修改
git pull <remote_name> <remote_branch>
# 将修改推送到远程仓库
git push <remote_name> <local_branch>
# 移除远程仓库
git remote rm <remote_name>
# 显示指定远程仓库信息
git remote show <remote_name>
# 显示远程仓库中的追踪分支
git remote show <remote_name> --verbose

# --------------- Advanced -------------------

# 从远程仓库中获取更新
git remote update
# 将修改强制推送到远程仓库，覆盖远程记录
git push --force <remote_name> <local_branch>
# 将标记推送到远程仓库
git push --tags <remote_name>
# 重命名远程仓库
git remote rename <old_name> <new_name>
# 修改远程仓库的URL
git remote set-url <name> <new_url>
# 删除旧的远程追踪分支
git remote prune <remote_name>
# 列出已合并到当前分支中的远程分支
git branch -r --merged
# 列出还未合并到当前分支的远程分支
git branch -r --no-merged
# 从远程仓库获取更新并删除废弃的远程追踪分支
git fetch -p

# 追踪远程分支并设置本地分支自动与之同步
git branch --track <branch_name> <remote_name>/<remote_branch>
# 设置本地现有分支以追踪远程分支
git branch -u <remote_name>/<remote_branch>
# 将分支推送到远程仓库并设置其追踪远程分支
git push -u <remote_name> <local_branch>
# 删除本地分支与远程分支的追踪关联
git branch --unset-upstream <branch_name>

提交历史

# 显示提交历史
git log
# 显示浓缩的提交历史
git log --oneline
# 显示分支提交历史
git log --graph
# 按作者过滤提交历史
git log --author=<author_name>
# 显示指定日期后的提交历史
git log --since=<date>
# 显式指定日期前的提交历史
git log --until=<date>

标签

# 列出标签
git tag
# 在指定的提交上创建标签
git tag <tag_name> <commit_id>
# 创建带有消息的带注释标签
git tag -a <tag_name> -m "tag message"
# 删除指定标签
git tag -d <tag_name>
# 删除指定的远程标签
git push <remote_name> --delete <tag_name>
# 显示指定的标签相关信息
git show <tag_name>

储藏

# 在工作树中临时保存更改
git stash save "stash message"
# 列出储藏
git stash list
# 应用指定储藏中的更改
git stash apply <stash>
# 移除指定储藏
git stash drop <stash>
# 移除所有储藏
git stash clear

筛选

# 将指定的提交从一个分支应用到另一个分支
git cherry-pick <commit_id>

提交管理

# Modify the latest commit
git commit --amend
# Create a new commit that undoes changes from a previous commit
git revert <commit_id>
# Discard changes and move HEAD to a specific commit
git reset --hard <commit_id>
# Move HEAD to a specific commit, but preserve staged changes
git reset --soft <commit_id>
# Show a record of all changes made to the local repository head
git reflog

子模块、子树和高级子模块

# 向当前仓库中添加一个子模块
git submodule add <repository_url> <path>
# 递归地初始化和更新所有子模块
git submodule update --init --recursive
# 向当前仓库中添加一个子树
git subtree add --prefix=<path> <repository_url>
# 初始化仓库中的子模块
git submodule init
# 将子模块更新到最近提交的版本
git submodule update
# 在每个子模块中执行指定的命令
git submodule foreach <command>
# 注销子模块
git submodule deinit <path>

钩子和自动化，以及Diff和合并工具

# 定位Git仓库中的hooks目录（通常在.git/hooks/)
git hooks
# 可添加至hooks目录中的特定钩子的脚本名称
pre-commit, post-commit, pre-push, post-merge, etc.
# 使钩子脚本可执行，确保在必要时被触发
chmod +x <hook_script>

使用补丁

# 为一次指定的提交生成补丁文件
git format-patch <commit_id>
# 对当前分支使用补丁
git apply <patch_file>
# 通过"git am" (apply mailbox) 命令使用补丁
git am <patch_file>

协作

# 生成包含两次提交之间更改的请求-拉取概述
git request-pull <start_commit> <end_commit> <url>
# 总结提交历史，列出作者和他们的贡献
git shortlog
# 列出Git追踪的文件
git ls-files
# 在Git追踪的文件中搜索指定的模式
git grep <pattern>

分割、调试和性能问题

# 开始一个平分会话来查找引入漏洞的提交
git bisect start
# 标记提交为“坏”，表示包含漏洞
git bisect bad <commit_id>
# 标记提交为“良好”，表示未包含漏洞
git bisect good <commit_id>
# 结束平分会话并返回到原始分支/提交
git bisect reset
# 验证Git仓库的完整性
git fsck
# 运行垃圾收集以优化仓库的性能
git gc
# 移除未追踪的文件和目录（谨慎使用）
git clean -df

技巧和窍门

# 交互式地选择要添加的文件
git add -p

# 显示指定文件的提交历史和相关补丁
git log -p <file_name>
# 自定义Git日志输出格式
git log --pretty=format:"%h - %an, %ar : %s"
# 在提交消息中查找文本（适用于定位特定变更）
git log --grep="<text>"
# 快速查看自上次提交以来的工作目录的更改
git diff --stat
# 显示带装饰的分支历史记录，以查看分支分裂或合并的位置
git log --oneline --decorate --graph

# 将更改保存在工作树中，包括未追踪的文件
git stash save -u
# 创建一个空提交，测试分支保护规则时很有用
git commit --allow-empty -m "Empty commit message"

# 设置Git输出页少于一个屏幕时退出，显示后不清除屏幕
git config --global core.pager 'less -RFX'
# 使用Git的自动校正功能来修复输错的命令
git config --global help.autocorrect 1
# 列出Git命令的别名
git config --get-regexp alias

# 以不实际合并分支的形式执行合并操作
git merge --no-commit --no-ff <branch_name>
# 显示仓库结构的树状表示
git ls-tree --name-only -r -t HEAD
